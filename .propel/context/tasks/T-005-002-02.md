## Description

Implement Excel export using SheetJS (xlsx library) with multi-sheet workbooks, formatted headers, and number formatting.

---

## Acceptance Criteria

- [ ] SheetJS (xlsx) library integrated
- [ ] Client-side Excel generation (no server round-trip)
- [ ] Multi-sheet workbook:
  - Overview worksheet
  - Separate worksheets per entity category
- [ ] Formatted headers: Bold, blue background (#3B82F6), white text
- [ ] Frozen pane (header row)
- [ ] Auto-sized columns to fit content
- [ ] Number formatting: Currency `$#,##0.00`, Percentages `0.00%`
- [ ] Filename: `{document_name}_entities_{YYYY-MM-DD}.xlsx`
- [ ] Empty categories produce empty sheet with headers

## Technical Implementation

```tsx
import * as XLSX from 'xlsx';

const exportToExcel = (data: EntityData, documentName: string) => {
  const workbook = XLSX.utils.book_new();
  
  // Overview sheet
  const overviewData = [
    ['Document', documentName],
    ['Export Date', new Date().toISOString()],
    ['Total Fields', countFields(data)],
  ];
  const overviewSheet = XLSX.utils.aoa_to_sheet(overviewData);
  XLSX.utils.book_append_sheet(workbook, overviewSheet, 'Overview');
  
  // Category sheets
  Object.entries(data.categories).forEach(([category, entities]) => {
    const sheetData = [['Field Name', 'Extracted Value']];
    entities.forEach(entity => {
      sheetData.push([entity.field, entity.value || '']);
    });
    const sheet = XLSX.utils.aoa_to_sheet(sheetData);
    
    // Style header row
    if (sheet['!ref']) {
      sheet['A1'].s = { fill: { fgColor: { rgb: '3B82F6' } }, font: { bold: true, color: { rgb: 'FFFFFF' } } };
      sheet['B1'].s = { fill: { fgColor: { rgb: '3B82F6' } }, font: { bold: true, color: { rgb: 'FFFFFF' } } };
    }
    
    XLSX.utils.book_append_sheet(workbook, sheet, category.slice(0, 31));
  });
  
  const date = new Date().toISOString().split('T')[0];
  XLSX.writeFile(workbook, `${documentName}_entities_${date}.xlsx`);
};
```
