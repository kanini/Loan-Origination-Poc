## Technical Implementation

```tsx
import jsPDF from 'jspdf';
import autoTable from 'jspdf-autotable';

const exportToPDF = (data: EntityData, documentName: string) => {
  const doc = new jsPDF();
  
  // Header
  doc.setFontSize(20);
  doc.text('Entity Extraction Report', 14, 22);
  
  doc.setFontSize(12);
  doc.text(`Document: ${documentName}`, 14, 32);
  doc.text(`Generated: ${new Date().toLocaleDateString()}`, 14, 40);
  
  let yPos = 50;
  
  // Entity categories
  Object.entries(data.categories).forEach(([category, entities]) => {
    if (yPos > 250) {
      doc.addPage();
      yPos = 20;
    }
    
    doc.setFontSize(14);
    doc.text(category, 14, yPos);
    yPos += 10;
    
    autoTable(doc, {
      startY: yPos,
      head: [['Field', 'Value']],
      body: entities.map(e => [e.field, e.value || '--']),
      margin: { left: 14 },
    });
    
    yPos = (doc as any).lastAutoTable.finalY + 15;
  });
  
  const date = new Date().toISOString().split('T')[0];
  doc.save(`${documentName}_report_${date}.pdf`);
};
```
